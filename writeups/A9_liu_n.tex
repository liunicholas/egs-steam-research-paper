% ── SLIDE 1: MOTIVATION & RESEARCH QUESTION ──────────────────────────────────
\begin{frame}{Motivation \& Research Question}

\begin{columns}[T] 
\column{0.56\textwidth}
  \textbf{Motivating example}
  \begin{itemize}
    \item Epic Games Store (EGS) has given away \alert{over 400 games} since 2018 --- its primary differentiation lever against Steam
    \item Standard theory predicts a \alert{zero-price substitute} on a rival platform diverts players \emph{away} from Steam
    \item But in January 2026, New Blood CEO Dave Oshry reported that giving away \textit{Blood West} on EGS caused Steam sales to surge \alert{$\sim$200\%} the same day
  \end{itemize}

  \vspace{0.7em}
  \textbf{Research question}
  \begin{block}{}
    Does an EGS free giveaway \emph{increase} or \emph{decrease} Steam player engagement for the same game --- and how does this evolve over time?
  \end{block}

\column{0.42\textwidth}
  \textbf{Two competing mechanisms}
  \vspace{0.4em}

  \textbf{Advertising:} Giveaway generates media buzz $\rightarrow$ players flock to Steam (primary discovery/community layer) $\rightarrow$ \alert{positive} short-run spillover

  \vspace{0.6em}
  \textbf{Substitution:} Free copy on Epic gradually diverts playtime away from Steam $\rightarrow$ \alert{negative} medium-run effect

  \vspace{0.6em}
  \textit{Both may operate, but at different horizons.}
\end{columns}

\end{frame}

% ── SLIDE 3: IDENTIFICATION STRATEGY ─────────────────────────────────────────
\begin{frame}{Identification Strategy}

\begin{columns}[T]
\column{0.50\textwidth}
  \textbf{Design: Staggered DiD Event Study}
  \begin{itemize}
    \item Treatment = game $i$'s first EGS giveaway, staggered across \alert{80 calendar months} (Dec 2018 -- late 2025)
    \item \textbf{Control group:} not-yet-treated games --- games awaiting future giveaways serve as controls for currently treated games
    \item No external never-treated group used
  \end{itemize}

  \vspace{0.5em}
  \textbf{Key assumption (parallel trends)}

  In absence of treatment, log player trajectories would trend identically after conditioning on game \& calendar-month FE.

\column{0.48\textwidth}
  \textbf{Four Estimators}

  \vspace{0.3em}
  \begin{enumerate}
    \item \textbf{TWFE} --- game + calendar-month FE; event window $k \in [-12,12]$
    \item \textbf{LP-DiD} (Dub\'e et al., 2023) --- local projections using only not-yet-treated controls at each $k$
    \item \textbf{Stacked DiD} (Cengiz et al., 2019) --- cohort-specific balanced panels
    \item \textbf{CS-ATT} (Callaway \& Sant'Anna, 2021) --- cohort-specific ATTs
  \end{enumerate}

  \vspace{0.4em}
  \textbf{Randomization placebo:} 200 permutations of treatment dates to validate the $k=0$ spike
\end{columns}

\end{frame}

% ── SLIDE 4: RESULTS ─────────────────────────────────────────────────────────
\begin{frame}{Results: Two-Phase Dynamic}

\begin{columns}[T]
\column{0.55\textwidth}
  \includegraphics[width=\linewidth]{../output/fig_estimator_comparison.png}

  \vspace{-0.2em}
  {\small TWFE, LP-DiD, and Stacked DiD. Game + calendar-month FE. $N = 38{,}641$; 446 games. Reference: $k=-1$.}

\column{0.43\textwidth}
  \textbf{Phase 1 --- Advertising spike}
  \begin{itemize}
    \item \alert{+10 to +14\%} more Steam players at $k=0$ across all estimators
    \item Randomization placebo $p < 0.001$
  \end{itemize}

  \vspace{0.5em}
  \textbf{Phase 2 --- Medium-run decline}
  \begin{itemize}
    \item All estimators turn negative by $k=2$; significant from $k=5$ onward
    \item \alert{$0$ to $-23\%$} below baseline by $k=12$
    \item Partly reflects \alert{pre-existing lifecycle decline}
  \end{itemize}
\end{columns}

\end{frame}

% ── SLIDE 5: ESTIMATOR COMPARISON (OPTIONAL / can skip for time) ──────────────
\begin{frame}{Robustness: Why Do Estimators Differ?}

\begin{columns}[T]
\column{0.54\textwidth}
  \textbf{They differ in \emph{which} games serve as the counterfactual.}

  \vspace{0.5em}
  \textbf{TWFE} --- control: all other game-months. Game FE removes levels but not \emph{trends} $\Rightarrow$ lifecycle decline leaks in, pulling long-run \alert{too negative}.

  \vspace{0.5em}
  \textbf{LP-DiD} --- control: games not yet treated at $k$. DiD cancels common trends, but controls decline \emph{faster} than treated at $k=0$ $\Rightarrow$ \alert{inflates $k=0$}.

  \vspace{0.5em}
  \textbf{Stacked DiD} --- control: matched not-yet-treated on each cohort's own trends. Absorbs both biases $\Rightarrow$ \alert{cleanest} read; medium-run $\approx 0$.

\column{0.44\textwidth}

  \begin{tabular}{lccc}
    \toprule
    $k$ & TWFE & LP-DiD & Stacked \\
    \midrule
    $0$  & $+0.101^{***}$ & $+0.131^{***}$ & $+0.101^{***}$ \\
    $1$  & $+0.045^{*}$  & $+0.099^{***}$ & $+0.049^{**}$ \\
    $2$  & $-0.063^{**}$ & $\phantom{+}0.000$ & $-0.040^{*}$ \\
    $5$  & $-0.145^{***}$ & $-0.064^{*}$ & $-0.097^{***}$ \\
    $10$ & $-0.199^{***}$ & $-0.121^{***}$ & $-0.061^{**}$ \\
    $12$ & $-0.267^{***}$ & $-0.105^{**}$ & $\phantom{-}0.016$ \\
    \bottomrule
  \end{tabular}

  \vspace{0.3em}
  \textit{\small $^*p<0.05$, $^{**}p<0.01$, $^{***}p<0.001$. SE clustered by game.}

  \vspace{0.4em}
  \textit{\small All three agree on the $k=0$ spike and reversal.}
\end{columns}

\end{frame}

% ── SLIDE 6: TAKEAWAYS & CAVEATS ─────────────────────────────────────────────
\begin{frame}{Takeaways, Caveats \& Future Work}

\begin{columns}[T]
\column{0.52\textwidth}
  \textbf{Main takeaway:} Advertising spike (\alert{$+10$ to $+14\%$} at $k=0$); medium-run reversal spans \alert{$0$ to $-23\%$} across estimators. Stacked-DiD path is near zero by $k=12$ $\Rightarrow$ TWFE/LP-DiD medium-run negatives are best read as \alert{upper bounds} on causal substitution.

  \vspace{0.5em}
  \textbf{Developers:} Fee + Steam spike are both positive; player migration to Epic is not a developer cost.

  \smallskip
  \textbf{Steam:} Net positive --- spike accrues to Steam; decline is largely pre-existing lifecycle.

  \smallskip
  \textbf{Epic:} Open question --- whether EGS user acquisition offsets the licensing cost is unknown.

\column{0.46\textwidth}
  \textbf{Caveats}
  \begin{itemize}
    \item \textbf{Lifecycle selection} --- games are already declining on Steam
    \item \textbf{Players $\neq$ purchases} --- concurrent player counts, not sales
    \item \textbf{Genre heterogeneity} --- pooled estimate mixes single-player and multiplayer
  \end{itemize}

  \vspace{0.4em}
  \textbf{Future work}
  \begin{itemize}
    \item Single-player vs.\ multiplayer heterogeneity
    \item Steam sales volume effects
    \item EGS-side engagement/sales data
  \end{itemize}
\end{columns}

\end{frame}
